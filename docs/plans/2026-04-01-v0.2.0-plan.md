```python
class EngineNotInstalled(RequestException):
    """引擎依赖未安装异常。

    当用户指定的引擎对应的第三方库未安装时抛出，
    异常消息中包含安装命令以引导用户安装。

    Attributes:
        engine_name: 引擎名称。
        install_package: 安装包名称（含 extras）。
    """

    def __init__(self, engine_name: str, install_package: str):
        """初始化引擎未安装异常。

        Args:
            engine_name: 引擎名称，如 "aiohttp"、"curl-cffi"。
            install_package: 安装包名称，如 "hs-net[aiohttp]"。
        """
        self.engine_name = engine_name
        self.install_package = install_package
        super().__init__(
            "EngineNotInstalled",
            f"引擎 {engine_name} 需要额外安装依赖: pip install {install_package}",
        )
```

```python
def _resolve_async_engine_cls(engine: str | EngineEnum | type[EngineBase]) -> type[EngineBase]:
    """根据引擎标识获取异步引擎类。

    Args:
        engine: 引擎标识，支持字符串、EngineEnum 枚举或 EngineBase 子类。

    Returns:
        对应的异步引擎类。

    Raises:
        ValueError: 当引擎标识不被支持时抛出。
    """
    if isinstance(engine, type) and issubclass(engine, EngineBase):
        return engine

    engine_value = engine.value if isinstance(engine, EngineEnum) else engine

    if engine_value == "requests":
        raise ValueError("requests 引擎不支持异步，请使用 SyncNet 或选择其他引擎")

    engine_map = _get_async_engine_map()
    if engine_value in engine_map:
        return engine_map[engine_value]

    raise ValueError(f"不支持的异步引擎: {engine_value}")
```

```python
"""模块级快捷函数，无需实例化即可发起 HTTP 请求。

每次调用会创建一个临时客户端，执行请求后自动关闭。
适用于简单的一次性请求场景，需要复用连接时请使用 Net / SyncNet。

    method: str,
    url: str,
    *,
    engine: str | None = None,
    **kwargs: Any,
) -> Response:
    """发起异步 HTTP 请求（临时客户端，用完即关）。

    Args:
        method: HTTP 方法（GET、POST 等）。
        url: 请求目标 URL。
        engine: 引擎名称，默认使用 httpx。
        **kwargs: 传递给 Net.request() 的其他参数

    Returns:
        Response 响应对象。

    Example::

        resp = await hs_net.request("GET", "https://example.com")
    """
    from hs_net.client import Net

    client = Net(engine=engine) if engine else Net()
    try:
        return await client.request(method, url, **kwargs)
    finally:
        await client.close()
```
